## 安装使用
```
npm install @position/vuem --save-dev
```
## 全局配置

```js
import Vuem, { VueDialog, VueLoading, VueToast} from '@position/vuem';

Vue.use(Vuem);

Vue.use(VueDialog, {
  container: '#app',
});

Vue.use(VueLoading, {
  container: '#app',
});

Vue.use(VueToast, {
  container: '#app',
});
```
